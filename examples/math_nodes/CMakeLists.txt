cmake_minimum_required(VERSION 3.2)
project(nodal-example-math-nodes CXX C)

if(NOT BUILDING_NODAL)
    list(APPEND CMAKE_MODULE_PATH "${PROJECT_SOURCE_DIR}/../../cmake")

    include(GNUInstallDirs)
    include(ConfigureSourceDefinitions)

    find_package(nodal REQUIRED)
endif()

set(SOURCES
    node_factory.cpp
    compiler.cpp
    main.cpp

    nodes/input.cpp
    nodes/output.cpp
    nodes/math.cpp
    nodes/random.cpp
)

set(HEADERS
    node.h
    node_factory.h
    context.h
    compiler.h

    nodes/all.h
    nodes/input.h
    nodes/output.h
    nodes/math.h
    nodes/random.h
)

configure_source_definitions(
    "__NODAL_EXAMPLE_MATH_NODES_BUILDING_@id@__"
    ${SOURCES})

add_executable(math-nodes ${SOURCES} ${HEADERS})

target_compile_features(math-nodes PUBLIC cxx_std_14)
target_link_libraries(math-nodes nodal)

set_property(TARGET math-nodes
             PROPERTY INTERPROCEDURAL_OPTIMIZATION TRUE)
